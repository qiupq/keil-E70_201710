00054 static void AlignedCopy(uint8_t * *src,
00055                         uint8_t * *dst,
00056                         uint32_t len)
00057 {
00058     uint32_t *src32, *dst32;
00059     src32 = (uint32_t *)(void *)*src;
00060     dst32 = (uint32_t *)(void *)*dst;
00061 
00062     for (; len > 0; len -= 8) {
00063         *dst32 ++ = *src32 ++;
00064         *dst32 ++ = *src32 ++;
00065     }
