00066      * \brief Setup the microcontroller system.
00067      * Initialize the System and update the SystemFrequency variable.

00069     void SystemInit(void)
00070 {
00071     uint32_t read_MOR;
00072     /* Set FWS according to SYS_BOARD_MCKR configuration */
00073     EFC->EEFC_FMR = EEFC_FMR_FWS(5);
00074 
00075     /* Before switching MAIN OSC on external crystal : enable it and don't
00076      * disable at the same time RC OSC in case of if MAIN OSC is still using RC
00077      * OSC
00078      */
00079 
00080     read_MOR = PMC->CKGR_MOR;
00081     /* enable external crystal - enable RC OSC */
00082     read_MOR |= (CKGR_MOR_KEY_PASSWD | CKGR_MOR_XT32KFME);
00083     PMC->CKGR_MOR = read_MOR;
00084 
00085     /* Initialize main oscillator */
00086     if (!(PMC->CKGR_MOR & CKGR_MOR_MOSCSEL)) {
00087         PMC->CKGR_MOR = CKGR_MOR_KEY_PASSWD | SYS_BOARD_OSCOUNT | CKGR_MOR_MOSCRCEN |
00088                         CKGR_MOR_MOSCXTEN;
00089 
00090         while (!(PMC->PMC_SR & PMC_SR_MOSCXTS)) {
00091         }
00092     }
00093 
00094     /* Switch to 3-20MHz Xtal oscillator */
00095     PMC->CKGR_MOR = CKGR_MOR_KEY_PASSWD | SYS_BOARD_OSCOUNT | CKGR_MOR_MOSCRCEN |
00096                     CKGR_MOR_MOSCXTEN | CKGR_MOR_MOSCSEL;
00097 
00098     while (!(PMC->PMC_SR & PMC_SR_MOSCSELS)) {
00099     }
00100 
00101     PMC->PMC_MCKR = (PMC->PMC_MCKR & ~(uint32_t)PMC_MCKR_CSS_Msk) |
00102                     PMC_MCKR_CSS_MAIN_CLK;
00103 
00104     while (!(PMC->PMC_SR & PMC_SR_MCKRDY)) {
00105     }
00106 
00107     /* Initialize PLLA */
00108     PMC->CKGR_PLLAR = SYS_BOARD_PLLAR;
00109 
00110     while (!(PMC->PMC_SR & PMC_SR_LOCKA)) {
00111     }
00112 
00113     /* Switch to main clock: DO NOT modify MDIV and CSS feild at the same access */
00114     PMC->PMC_MCKR = (PMC->PMC_MCKR & ~(uint32_t)PMC_MCKR_MDIV_Msk) |
00115                     SYS_BOARD_MCKR_MDIV;
00116     PMC->PMC_MCKR = (SYS_BOARD_MCKR & ~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK;
00117 
00118     while (!(PMC->PMC_SR & PMC_SR_MCKRDY)) {
00119     }
00120 
00121     /* Switch to PLLA */
00122     PMC->PMC_MCKR = SYS_BOARD_MCKR;
00123 
00124     while (!(PMC->PMC_SR & PMC_SR_MCKRDY)) {
00125     }
00126 
00127     SystemCoreClock = CHIP_FREQ_CPU_MAX;

00130 void SystemCoreClockUpdate(void)
00131 {
00132     /* Determine clock frequency according to clock register values */
00133     switch (PMC->PMC_MCKR & (uint32_t) PMC_MCKR_CSS_Msk) {
00134     case PMC_MCKR_CSS_SLOW_CLK: /* Slow clock */
00135         if (SUPC->SUPC_SR & SUPC_SR_OSCSEL)
00136             SystemCoreClock = CHIP_FREQ_XTAL_32K;
00137         else
00138             SystemCoreClock = CHIP_FREQ_SLCK_RC;
00139 
00140         break;
00141 
00142     case PMC_MCKR_CSS_MAIN_CLK: /* Main clock */
00143         if (PMC->CKGR_MOR & CKGR_MOR_MOSCSEL)
00144             SystemCoreClock = CHIP_FREQ_XTAL_12M;
00145         else {
00146             SystemCoreClock = CHIP_FREQ_MAINCK_RC_4MHZ;
00147 
00148             switch (PMC->CKGR_MOR & CKGR_MOR_MOSCRCF_Msk) {
00149             case CKGR_MOR_MOSCRCF_4_MHz:
00150                 break;
00151 
00152             case CKGR_MOR_MOSCRCF_8_MHz:
00153                 SystemCoreClock *= 2U;
00154                 break;
00155 
00156             case CKGR_MOR_MOSCRCF_12_MHz:
00157                 SystemCoreClock *= 3U;
00158                 break;
00159 
00160             default:
00161                 break;
00162             }
00163         }
00164 
00165         break;
00166 
00167     case PMC_MCKR_CSS_PLLA_CLK: /* PLLA clock */
00168         if (PMC->CKGR_MOR & CKGR_MOR_MOSCSEL)
00169             SystemCoreClock = CHIP_FREQ_XTAL_12M;
00170         else {
00171             SystemCoreClock = CHIP_FREQ_MAINCK_RC_4MHZ;
00172 
00173             switch (PMC->CKGR_MOR & CKGR_MOR_MOSCRCF_Msk) {
00174             case CKGR_MOR_MOSCRCF_4_MHz:
00175                 break;
00176 
00177             case CKGR_MOR_MOSCRCF_8_MHz:
00178                 SystemCoreClock *= 2U;
00179                 break;
00180 
00181             case CKGR_MOR_MOSCRCF_12_MHz:
00182                 SystemCoreClock *= 3U;
00183                 break;
00184 
00185             default:
00186                 break;
00187             }
00188         }
00189 
00190         if ((uint32_t) (PMC->PMC_MCKR & (uint32_t) PMC_MCKR_CSS_Msk) ==
00191              PMC_MCKR_CSS_PLLA_CLK) {
00192             SystemCoreClock *= ((((PMC->CKGR_PLLAR) & CKGR_PLLAR_MULA_Msk) >>
00193                                  CKGR_PLLAR_MULA_Pos) + 1U);
00194             SystemCoreClock /= ((((PMC->CKGR_PLLAR) & CKGR_PLLAR_DIVA_Msk) >>
00195                                  CKGR_PLLAR_DIVA_Pos));
00196         }
00197 
00198         break;
00199 
00200     default:
00201         break;
00202     }
00203 
00204     if ((PMC->PMC_MCKR & PMC_MCKR_PRES_Msk) == PMC_MCKR_PRES_CLK_3)
00205         SystemCoreClock /= 3U;
00206     else
00207         SystemCoreClock >>= ((PMC->PMC_MCKR & PMC_MCKR_PRES_Msk) >> PMC_MCKR_PRES_Pos);
00208 }

00212 void system_init_flash(uint32_t ul_clk)
00213 {
00214     /* Set FWS for embedded Flash access according to operating frequency */
00215     if (ul_clk < CHIP_FREQ_FWS_0)

00220         else {

00222                 EFC->EEFC_FMR = EEFC_FMR_FWS(2) | EEFC_FMR_CLOE;
00223             else {
00224                 if (ul_clk < CHIP_FREQ_FWS_3)
00225                     EFC->EEFC_FMR = EEFC_FMR_FWS(3) | EEFC_FMR_CLOE;
00226                 else {
