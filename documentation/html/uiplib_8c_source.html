00043 uiplib_ipaddrconv(char *addrstr, unsigned char *ipaddr)
00044 {
00045   unsigned char tmp;
00046   char c;
00047   unsigned char i, j;
00048 
00049   tmp = 0;
00050 
00051   for(i = 0; i < 4; ++i) {
00052     j = 0;
00053     do {
00054       c = *addrstr;
00055       ++j;
00056       if(j > 4) {
00057     return 0;
00058       }
00059       if(c == '.' || c == 0) {
00060     *ipaddr = tmp;
00061     ++ipaddr;
00062     tmp = 0;
00063       } else if(c >= '0' && c <= '9') {
00064     tmp = (tmp * 10) + (c - '0');
00065       } else {
00066     return 0;
00067       }
00068       ++addrstr;
00069     } while(c != '.' && c != 0);
00070   }
00071   return 1;
00072 }
