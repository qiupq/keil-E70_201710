00051 /* IP addresses and port numbers are expected to be in
00052  * the same byte order as in the corresponding pcb.
00053  */
00054 /** This struct includes everything that is necessary to execute a function
00055     for a netconn in another thread context (mainly used to process netconns
00056     in the tcpip_thread context to be thread safe). */
00057 struct api_msg_msg {
00058   /** The netconn which to process - always needed: it includes the semaphore
00059       which is used to block the application thread until the function finished. */
00060   struct netconn *conn;
00061   /** Depending on the executed function, one of these union members is used */
00062   union {
00063     /** used for do_send */
00064     struct netbuf *b;
00065     /** used for do_newconn */
00066     struct {
00067       u8_t proto;
00068     } n;
00069     /** used for do_bind and do_connect */
00070     struct {
00071       struct ip_addr *ipaddr;
00072       u16_t port;
00073     } bc;
00074     /** used for do_getaddr */
00075     struct {
00076       struct ip_addr *ipaddr;
00077       u16_t *port;
00078       u8_t local;
00079     } ad;
00080     /** used for do_write */
00081     struct {
00082       const void *dataptr;
00083       size_t len;
00084       u8_t apiflags;
00085     } w;
00086     /** used for do_recv */
00087     struct {
00088       u16_t len;
00089     } r;
00090 #if LWIP_IGMP
00091     /** used for do_join_leave_group */
00092     struct {
00093       struct ip_addr *multiaddr;
00094       struct ip_addr *interface;
00095       enum netconn_igmp join_or_leave;
00096     } jl;
00097 #endif /* LWIP_IGMP */
00098 #if TCP_LISTEN_BACKLOG
00099     struct {
00100       u8_t backlog;
00101     } lb;
00102 #endif /* TCP_LISTEN_BACKLOG */
00103   } msg;
00104 };

00106 /** This struct contains a function to execute in another thread context and
00107     a struct api_msg_msg that serves as an argument for this function.
00108     This is passed to tcpip_apimsg to execute functions in tcpip_thread context. */
00109 struct api_msg {
00110   /** function to execute in tcpip_thread context */
00111   void (* function)(struct api_msg_msg *msg);
00112   /** arguments for this function */
00113   struct api_msg_msg msg;
00114 };
