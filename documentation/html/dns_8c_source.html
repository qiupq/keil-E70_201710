00044  * The lwIP DNS resolver functions are used to lookup a host name and
00045  * map it to a numerical IP address. It maintains a list of resolved
00046  * hostnames that can be queried with the dns_lookup() function.
00047  * New hostnames can be resolved using the dns_query() function.
00048  *
00049  * The lwIP version of the resolver also adds a non-blocking version of
00050  * gethostbyname() that will work with a raw API application. This function
00051  * checks for an IP address string first and converts it if it is valid.
00052  * gethostbyname() then does a dns_lookup() to see if the name is 
00053  * already in the table. If so, the IP is returned. If not, a query is 
00054  * issued and the function returns with a ERR_INPROGRESS status. The app
00055  * using the dns client must then go into a waiting state.
00056  *
00057  * Once a hostname has been resolved (or found to be non-existent),
00058  * the resolver code calls a specified callback function (which 
00059  * must be implemented by the module that uses the resolver).
00060  */

00492  * to make sure an answer from the DNS server matches the current dns_table
00493  * entry (otherwise, answers might arrive late for hostname not on the list

00502 {
00503   unsigned char n;
00504 
00505   do {
00506     n = *response++;
00507     /** @see RFC 1035 - 4.1.4. Message compression */

00509       /* Compressed name */
00510       break;
00511     } else {
00512       /* Not compressed name */
00513       while (n > 0) {
00514         if ((*query) != (*response)) {
00515           return 1;
00516         }
00517         ++response;
00518         ++query;
00519         --n;
00520       };
00521       ++query;
00522     }
00523   } while (*response != 0);
00524

00634 /**
00635  * dns_check_entry() - see if pEntry has not yet been queried and, if so, sends out a query.
00636  * Check an entry in the dns_table:
00637  * - send out query for new entries
00638  * - retry old pending entries on timeout (also with different servers)
00639  * - remove completed entries from the table if their TTL has expired
00640  *
00641  * @param i index of the dns_table entry to check
00642  */
00643 static void
00644 dns_check_entry(u8_t i)
00645 {
00646   struct dns_table_entry *pEntry = &dns_table[i];
00647 
00648   LWIP_ASSERT("array index out of bounds", i < DNS_TABLE_SIZE);
00649 
00650   switch(pEntry->state) {
00651 
00652     case DNS_STATE_NEW: {
00653       /* initialize new entry */
00654       pEntry->state   = DNS_STATE_ASKING;
00655       pEntry->numdns  = 0;
00656       pEntry->tmr     = 1;
00657       pEntry->retries = 0;
00658       
00659       /* send DNS packet for this entry */
00660       dns_send(pEntry->numdns, pEntry->name, i);
00661       break;
00662     }
00663 
00664     case DNS_STATE_ASKING: {
00665       if (--pEntry->tmr == 0) {
00666         if (++pEntry->retries == DNS_MAX_RETRIES) {
00667           if ((pEntry->numdns+1<DNS_MAX_SERVERS) && (dns_servers[pEntry->numdns+1].addr!=0)) {
00668             /* change of server */
00669             pEntry->numdns++;
00670             pEntry->tmr     = 1;
00671             pEntry->retries = 0;
00672             break;
00673           } else {
00674             LWIP_DEBUGF(DNS_DEBUG, ("dns_check_entry: \"%s\": timeout\n", pEntry->name));
00675             /* call specified callback function if provided */
00676             if (pEntry->found)
00677               (*pEntry->found)(pEntry->name, NULL, pEntry->arg);
00678             /* flush this entry */
00679             pEntry->state   = DNS_STATE_UNUSED;
00680             pEntry->found   = NULL;
00681             break;
00682           }
00683         }
00684 
00685         /* wait longer for the next retry */
00686         pEntry->tmr = pEntry->retries;
00687 
00688         /* send DNS packet for this entry */
00689         dns_send(pEntry->numdns, pEntry->name, i);
00690       }
00691       break;
00692     }
00693 
00694     case DNS_STATE_DONE: {
00695       /* if the time to live is nul */
00696       if (--pEntry->ttl == 0) {
00697         LWIP_DEBUGF(DNS_DEBUG, ("dns_check_entry: \"%s\": flush\n", pEntry->name));
00698         /* flush this entry */
00699         pEntry->state = DNS_STATE_UNUSED;
00700         pEntry->found = NULL;
00701       }
00702       break;
00703     }
00704     case DNS_STATE_UNUSED:
00705       /* nothing to do */
00706       break;
00707     default:
00708       LWIP_ASSERT("unknown dns_table entry state:", 0);
00709       break;
00710   }
00711 }

00755     /* free pbuf and return */

00757   }
00758 
00759   /* is the dns message big enough ? */
00760   if (p->tot_len < (SIZEOF_DNS_HDR + SIZEOF_DNS_QUERY + SIZEOF_DNS_ANSWER)) {
00761     LWIP_DEBUGF(DNS_DEBUG, ("dns_recv: pbuf too small\n"));
00762     /* free pbuf and return */
00763     goto memerr1;
00764   }
00765 
00766 #if (DNS_USES_STATIC_BUF == 2)
00767   dns_payload = mem_malloc(p->tot_len);
00768   if (dns_payload == NULL) {
00769     LWIP_DEBUGF(DNS_DEBUG, ("dns_recv: mem_malloc error\n"));
00770     /* free pbuf and return */
00771     goto memerr1;
00772   }
00773 #endif /* (DNS_USES_STATIC_BUF == 2) */
00774 
00775   /* copy dns payload inside static buffer for processing */ 
00776   if (pbuf_copy_partial(p, dns_payload, p->tot_len, 0) == p->tot_len) {

00788            and the extrarr are simply discarded. */

00791 
00792         /* Check for error. If so, call callback to inform. */

00794           LWIP_DEBUGF(DNS_DEBUG, ("dns_recv: \"%s\": error in flags\n", pEntry->name));
00795           /* call callback to indicate error, clean up memory and return */
00796           goto responseerr;
00797         }
00798 
00799 #if DNS_DOES_NAME_CHECK
00800         /* Check if the name in the "question" part match with the name in the entry. */
00801         if (dns_compare_name((unsigned char *)(pEntry->name), (unsigned char *)dns_payload + SIZEOF_DNS_HDR) != 0) {
00802           LWIP_DEBUGF(DNS_DEBUG, ("dns_recv: \"%s\": response not match to query\n", pEntry->name));
00803           /* call callback to indicate error, clean up memory and return */
00804           goto responseerr;
00805         }
00806 #endif /* DNS_DOES_NAME_CHECK */
00807 
00808         /* Skip the name in the "question" part */
00809         pHostname = (char *) dns_parse_name((unsigned char *)dns_payload + SIZEOF_DNS_HDR) + SIZEOF_DNS_QUERY;

00817           if((ntohs(ans.type) == DNS_RRTYPE_A) && (ntohs(ans.class) == DNS_RRCLASS_IN) && (ntohs(ans.len) == sizeof(struct ip_addr)) ) {

00819             pEntry->ttl = ntohl(ans.ttl);

00821               pEntry->ttl = DNS_MAX_TTL;
00822             }
00823             /* read the IP address after answer resource record's header */
00824             MEMCPY( &(pEntry->ipaddr), (pHostname+SIZEOF_DNS_ANSWER), sizeof(struct ip_addr));
00825             LWIP_DEBUGF(DNS_DEBUG, ("dns_recv: \"%s\": response = ", pEntry->name));
00826             ip_addr_debug_print(DNS_DEBUG, (&(pEntry->ipaddr)));
00827             LWIP_DEBUGF(DNS_DEBUG, ("\n"));
00828             /* call specified callback function if provided */
00829             if (pEntry->found) {
00830               (*pEntry->found)(pEntry->name, &pEntry->ipaddr, pEntry->arg);
00831             }
00832             /* deallocate memory and return */
00833             goto memerr2;
00834           } else {
00835             pHostname = pHostname + SIZEOF_DNS_ANSWER + htons(ans.len);
00836           }
00837           --nanswers;
00838         }
00839         LWIP_DEBUGF(DNS_DEBUG, ("dns_recv: \"%s\": error in response\n", pEntry->name));
00840         /* call callback to indicate error, clean up memory and return */
00841         goto responseerr;
00842       }
00843     }
00844   }
00845

00853   }
00854   /* flush this entry */
00855   pEntry->state = DNS_STATE_UNUSED;
00856   pEntry->found = NULL;
00857 
00858 memerr2:

00861   mem_free(dns_payload);
00862 #endif /* (DNS_USES_STATIC_BUF == 2) */
00863 
00864 memerr1:

00870 /**
00871  * Queues a new hostname to resolve and sends out a DNS query for that hostname
00872  *
00873  * @param name the hostname that is to be queried
00874  * @param found a callback founction to be called on success, failure or timeout
00875  * @param callback_arg argument to pass to the callback function
00876  * @return @return a err_t return code.
00877  */
00878 static err_t
00879 dns_enqueue(const char *name, dns_found_callback found, void *callback_arg)
00880 {
00881   u8_t i;
00882   u8_t lseq, lseqi;
00883   struct dns_table_entry *pEntry = NULL;
00884 
00885   /* search an unused entry, or the oldest one */
00886   lseq = lseqi = 0;
00887   for (i = 0; i < DNS_TABLE_SIZE; ++i) {
00888     pEntry = &dns_table[i];
00889     /* is it an unused entry ? */
00890     if (pEntry->state == DNS_STATE_UNUSED)
00891       break;
00892 
00893     /* check if this is the oldest completed entry */
00894     if (pEntry->state == DNS_STATE_DONE) {
00895       if ((dns_seqno - pEntry->seqno) > lseq) {
00896         lseq = dns_seqno - pEntry->seqno;
00897         lseqi = i;
00898       }
00899     }
00900   }
00901 
00902   /* if we don't have found an unused entry, use the oldest completed one */
00903   if (i == DNS_TABLE_SIZE) {
00904     if ((lseqi >= DNS_TABLE_SIZE) || (dns_table[lseqi].state != DNS_STATE_DONE)) {
00905       /* no entry can't be used now, table is full */
00906       LWIP_DEBUGF(DNS_DEBUG, ("dns_enqueue: \"%s\": DNS entries table is full\n", name));
00907       return ERR_MEM;
00908     } else {
00909       /* use the oldest completed one */
00910       i = lseqi;
00911       pEntry = &dns_table[i];
00912     }
00913   }
00914 
00915   /* use this entry */
00916   LWIP_DEBUGF(DNS_DEBUG, ("dns_enqueue: \"%s\": use DNS entry %"U16_F"\n", name, (u16_t)(i)));
00917 
00918   /* fill the entry */
00919   pEntry->state = DNS_STATE_NEW;
00920   pEntry->seqno = dns_seqno++;
00921   pEntry->found = found;
00922   pEntry->arg   = callback_arg;
00923   strcpy(pEntry->name, name);
00924 
00925   /* force to send query without waiting timer */
00926   dns_check_entry(i);
00927 
00928   /* dns query is enqueued */
00929   return ERR_INPROGRESS;
00930 }

00945  * @param found a callback function to be called on success, failure or timeout (only if

00947  * @param callback_arg argument to pass to the callback function
00948  * @return a err_t return code.

00950 err_t

00952                   void *callback_arg)
00953 {
00954   /* not initialized or no valid server yet, or invalid addr pointer
00955    * or invalid hostname or invalid hostname length */
00956   if ((dns_pcb == NULL) || (addr == NULL) ||
00957       (!hostname) || (!hostname[0]) ||
00958       (strlen(hostname) >= DNS_MAX_NAME_LENGTH)) {
00959     return ERR_VAL;
00960   }
00961 
00962 #if LWIP_HAVE_LOOPIF
