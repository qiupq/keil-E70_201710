00170 /*
00171  * Inline versions of get/put char/short/long.
00172  * Pointer is advanced; we assume that both arguments
00173  * are lvalues and will already be in registers.
00174  * cp MUST be u_char *.
00175  */
00176 #define GETCHAR(c, cp) { \
00177     (c) = *(cp)++; \
00178 }
00179 #define PUTCHAR(c, cp) { \
00180     *(cp)++ = (u_char) (c); \
00181 }

00184 #define GETSHORT(s, cp) { \
00185     (s) = *(cp); (cp)++; (s) <<= 8; \
00186     (s) |= *(cp); (cp)++; \
00187 }
00188 #define PUTSHORT(s, cp) { \
00189     *(cp)++ = (u_char) ((s) >> 8); \
00190     *(cp)++ = (u_char) (s & 0xff); \
00191 }
00192 
00193 #define GETLONG(l, cp) { \
00194     (l) = *(cp); (cp)++; (l) <<= 8; \
00195     (l) |= *(cp); (cp)++; (l) <<= 8; \
00196     (l) |= *(cp); (cp)++; (l) <<= 8; \
00197     (l) |= *(cp); (cp)++; \
00198 }
00199 #define PUTLONG(l, cp) { \
00200     *(cp)++ = (u_char) ((l) >> 24); \
00201     *(cp)++ = (u_char) ((l) >> 16); \
00202     *(cp)++ = (u_char) ((l) >> 8); \
00203     *(cp)++ = (u_char) (l); \
00204 }

00350 /* Warning: Using PPPAUTHTYPE_ANY might have security consequences.
00351  * RFC 1994 says:
00352  *
00353  * In practice, within or associated with each PPP server, there is a
00354  * database which associates "user" names with authentication
00355  * information ("secrets").  It is not anticipated that a particular
00356  * named user would be authenticated by multiple methods.  This would
00357  * make the user vulnerable to attacks which negotiate the least secure
00358  * method from among a set (such as PAP rather than CHAP).  If the same
00359  * secret was used, PAP would reveal the secret to be used later with
00360  * CHAP.
00361  *
00362  * Instead, for each user name there should be an indication of exactly
00363  * one method used to authenticate that user name.  If a user needs to
00364  * make use of different authentication methods under different
00365  * circumstances, then distinct user names SHOULD be employed, each of
00366  * which identifies exactly one authentication method.
00367  *
00368  */
00369 enum pppAuthType {
00370     PPPAUTHTYPE_NONE,
00371     PPPAUTHTYPE_ANY,
00372     PPPAUTHTYPE_PAP,
00373     PPPAUTHTYPE_CHAP
00374 };
