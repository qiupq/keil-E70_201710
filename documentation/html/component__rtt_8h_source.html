00056 #define RTT_MR_RTC1HZ (0x1u << 24) /**< \brief (RTT_MR) Real-Time Clock 1Hz Clock Selection */

00058 #define RTT_AR_ALMV_Pos 0
00059 #define RTT_AR_ALMV_Msk (0xffffffffu << RTT_AR_ALMV_Pos) /**< \brief (RTT_AR) Alarm Value */
00060 #define RTT_AR_ALMV(value) ((RTT_AR_ALMV_Msk & ((value) << RTT_AR_ALMV_Pos)))
00061 /* -------- RTT_VR : (RTT Offset: 0x08) Value Register -------- */
00062 #define RTT_VR_CRTV_Pos 0
00063 #define RTT_VR_CRTV_Msk (0xffffffffu << RTT_VR_CRTV_Pos) /**< \brief (RTT_VR) Current Real-time Value */
00064 /* -------- RTT_SR : (RTT Offset: 0x0C) Status Register -------- */
